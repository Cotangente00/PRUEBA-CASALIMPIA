CREATE TABLE detalleVenta(
	id int primary key not null auto_increment,
  	idVenta int(11),
    idProducto int(11),
    cantidad tinyint(4)
);

CREATE TABLE persona(
    id int primary key not null auto_increment,
    numDocumento bigint(20),
    nombres varchar(200),
    apellidos varchar(200),
    sexo char(1),
    numCelular bigint(20)
);

CREATE TABLE producto(
    id int primary key not null auto_increment,
    nombre varchar(200),
    precio decimal(7.2)
);

CREATE TABLE venta(
    id int primary key not null auto_increment,
    fecha timestamp,
    idPersona tinyint(4),
    monto decimal(7.2)
);
